== Access ==
{{- if .Values.ingress.enabled }}

  SourceBridge is available at:
  {{- if .Values.global.tls.enabled }}
    https://{{ .Values.global.domain }}
  {{- else }}
    http://{{ .Values.global.domain }}
  {{- end }}

  API endpoint:
  {{- if .Values.global.tls.enabled }}
    https://{{ .Values.global.domain }}/api
  {{- else }}
    http://{{ .Values.global.domain }}/api
  {{- end }}
{{- else }}

  No ingress is enabled. Use port-forwarding to access SourceBridge:

  # Web UI (open http://localhost:3000 in your browser)
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "sourcebridge.fullname" . }}-web {{ .Values.web.service.port }}:{{ .Values.web.service.port }}

  # API server (http://localhost:8080)
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "sourcebridge.fullname" . }}-api {{ .Values.api.service.port }}:{{ .Values.api.service.port }}

  # Health check
  curl http://localhost:{{ .Values.api.service.port }}/healthz
{{- end }}
